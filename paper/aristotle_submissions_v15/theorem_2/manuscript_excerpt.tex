\begin{theorem}[Monotonicity]\label{thm:monotonicity}
If $(p_2, \bs_2)$ is a velocity-matched multiple of $(p_1, \bs_1)$---i.e., $p_2 = m \cdot p_1$ and $s_2^{(i)} \equiv m \cdot s_1^{(i)} \pmod{D_i}$---then the orbit partition under $(p_2, \bs_2)$ refines that under $(p_1, \bs_1)$, and $d^*(p_2, \bs_2) \leq d^*(p_1, \bs_1)$.
\end{theorem}
