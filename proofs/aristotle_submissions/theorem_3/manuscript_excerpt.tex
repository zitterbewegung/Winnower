\begin{theorem}[Stabilization]\label{thm:stabilization}
Let $\mathcal{C}$ be a finite candidate set.
Assume convergent orbit-class frequencies: $\hat{\theta}_j(T) \to \theta_j^*$ as $T \to \infty$.
Define the per-site NLL rate $\lambda_c = \frac{1}{k_c} \sum_{j} \Hb(\theta_j^*)$, and let $\lambda^* = \min_c \lambda_c$.
Then:
\begin{enumerate}
    \item[\textup{(i)}] $\NML(c, T) = \lambda_c \cdot N(T) + \frac{k_c}{2}\log_2 T + o(N(T))$.
    \item[\textup{(ii)}] Every candidate with $\lambda_c > \lambda^*$ is eventually excluded.
    \item[\textup{(iii)}] If $\lambda^*$ is achieved uniquely, selection locks to a single winner.
\end{enumerate}
\end{theorem}
